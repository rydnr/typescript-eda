export function Enable(adapter: any): ClassDecorator {
  return (target: any) => {
    if (!Reflect.hasMetadata('adapters', target)) {
      Reflect.defineMetadata('adapters', [], target);
    }
    const adapters = Reflect.getMetadata('adapters', target);
    adapters.push(adapter);
  };
}
